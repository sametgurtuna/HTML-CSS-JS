Switch Case ATM

using System; using System.Collections.Generic; using System.Linq; using System.Text; using System.Threading.Tasks; namespace ATMButSwitchCase { internal class Program { static void Main(string[] args) { /* 1- Bakiye Görüntüleme 2- Para Çek 3- Para Yatır 4- Çıkış Yap */ int bakiye = 5900; Console.WriteLine("ZortBank'a Hoşgeldiniz."); Console.WriteLine("Lütfen yapmak istediğiniz işlemi seçiniz."); Console.WriteLine("1- Bakiye Görüntüleme"); Console.WriteLine("2- Para Çek"); Console.WriteLine("3- Para Yatır"); Console.WriteLine("4- Çıkış Yap"); string secim = Console.ReadLine(); switch (secim) { case "1": Console.WriteLine("Bakiyeniz: " + bakiye); Console.ReadLine(); break; case "2": Console.WriteLine("Çekmek istediğiniz miktarı giriniz: "); int cekilecekMiktar = Convert.ToInt32(Console.ReadLine()); if (cekilecekMiktar > bakiye) { Console.WriteLine("Bakiyenizden fazla para çekemezsiniz "); Console.ReadLine(); } else { Console.WriteLine("Kalan Bakiyeniz" + (bakiye - cekilecekMiktar)); Console.ReadLine(); } break; case "3": Console.WriteLine("Yatırmak istediğiniz mikarı giriniz:"); int yatirilacakMiktar = Convert.ToInt32(Console.ReadLine()); Console.WriteLine("Yeni Bakiyeniz" + (bakiye + yatirilacakMiktar)); Console.ReadLine(); break; case "4": Console.WriteLine("Çıkış Yapılıyor..."); Console.WriteLine("İyi günler dileriz."); Console.ReadLine(); break; default: Console.WriteLine("Lütfen geçerli bir değer giriniz"); Console.ReadLine(); break; Console.ReadLine(); } } } }
